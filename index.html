    const fillArray = (start, end) => {
        let newArray = [];          
        if (start <= end) {
            for (let i = start; i <= end; i++) {
                newArray.push(i);
            }
        } else if (start > end) {
            for (let i = start; i >= end; i--) {
                newArray.push(i);
            }
        } else {
            throw "Start or End point are undefined or NaN";
        }
        return newArray;
    };
    fillArray(1,5)  //[1,2,3,4,5]
